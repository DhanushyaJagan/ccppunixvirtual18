  MENU

          2. Subtract 
          3. Multiply 
          4. Divide 
          5. Modulus 
Enter your choice: 1 
Enter your two numbers: 12 15 
Result: 27 
  
Continue? y

5.#include <stdio.h>
 

int checkArmstrong(int num){
    int tempNumber,rem,sum;
    tempNumber=num;
 
    sum=0;
    while(tempNumber!=0)
    {
        rem=tempNumber%10;
        sum=sum + (rem*rem*rem);
        tempNumber/=10;
    }  
  if(sum==num)  
      return 1;
  else
      return 0;
}
